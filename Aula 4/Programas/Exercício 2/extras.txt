		EXPORT __main
			
		AREA 	M_DADOS1, DATA, READWRITE ; READONLY = JA ESCREVE NA MEMORIA AUTOMATICAMENTE /// READWRITE = DADOS NA MEMORIA MANUALMENTE
m		SPACE	2
n 		SPACE	2

		AREA 	M_DADOS2, DATA, READWRITE		
mn1		SPACE	4 ; O MINIMO SERIA UM ESPACO DE 3, POREM 4 E MELHOR PARA VISUALIZAR OS RESULTADOS	
		AREA 	M_PROG, CODE, READONLY
__main
		LDR		r0,=m
		LDR		r1,=n
		LDR		r5,=mn1
LOOP		LDRH	r2,[r0]
		LDRH	r3,[r1]
		SUBS	r4,r2,r3
		STRH	r4,[r5]
		B		LOOP
		END
